window.addEventListener('wheel', function(e) {
  e.preventDefault();
  const scrollStep = 2000;
  let scrollAmount = 0;

  if (e.deltaY < 0) {
    scrollAmount = -scrollStep;
  } else {
    scrollAmount = scrollStep;
  }

  const currentPosition = window.scrollX;
  const targetPosition = currentPosition + scrollAmount;

  window.scrollTo({
    left: targetPosition,
    top: 0,
    behavior: "smooth"
  });
});
